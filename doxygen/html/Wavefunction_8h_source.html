34 /** \brief Calculate the model potentials.
35  *
36  * This function calculates the model potentials and returns \f$
37  * V_{\mathrm{mod}}(r) = V_{\mathrm{C}}(r) + V_{\mathrm{P}}(r) +
38  * V_{\mathrm{s.o.}}(r) \f$
39  *
40  * \f[
41  * \begin{aligned}
42  * V_{\mathrm{C}}(r) &= - \frac{e^2}{4\pi\varepsilon_0} \frac{1 +
43  * (Z-1)\mathrm{e}^{-\alpha_1 r} - r(\alpha_3+\alpha_4 r)\mathrm{e}^{-\alpha_2
44  * r}}{r} \\
45  * V_{\mathrm{P}}(r) &= -\frac{e^2}{(4\pi\varepsilon_0)^2}
46  * \frac{\alpha_d}{2 r^4} \left[ 1 - \mathrm{e}^{-(r/r_c)^6} \right] \\
47  * V_{\mathrm{s.o.}}(r) &= \frac{1}{2} \left( \frac{e^2}{4\pi\varepsilon_0}
48  * \right) \left( \frac{g_s}{2 m_e^2 c^2} \right)
49  * \frac{\boldsymbol{l}\cdot\boldsymbol{s}}{r^3} \end{aligned} \f]
50  *
51  * \param[in] qd Quantum defect data (parameters for the potentials)
52  * \param[in] x Radial position
53  * \returns Model potentials evaluted at position \p x with parameters \p qd
54  */

73 /** \brief %Numerov's method
74  *
75  * This class implements %Numerov's method using a couple of helper functions.
76  * %Numerov's method solves a differential equation of the form
77  * \f[
78  * y''(x) + g(x)y(x) = 0
79  * \f]
80  * The equation is solved by iteratively computing
81  * \f[
82  * y_{n+1} = \frac{\left( 2 + \frac{5 h^2}{6} g_n \right) y_n - \left( 1 +
83  * \frac{h^2}{12} g_{n-1} \right) y_{n-1}}{\left( 1 + \frac{h^2}{12} g_n
84  * \right)} \f]
85  *
86  * with \f$ y_n = y(x_n) \f$ and \f$ g_n = g(x_n) \f$. Clearly this equation
87  * needs two initial conditions \f$ y_0 \f$ and \f$ y_1 \f$. We integrate the
88  * equation from outside to inside. Since the wavefunction hs to decay to zero
89  * at infinity we choose \f$ y_0 = 0 \f$. Now depending on the number of knots
90  * of wavefunction we decide whether to set \f$ y_1 = \pm\varepsilon \f$.
91  */

105  *

220 /** \brief Find and return index
221  *
222  * Find a value in an Eigen matrix and return its index in the
223  * container to the caller. This is intended to work with
224  * Eigen::Vector or columns of Eigen::Matrix.
225  *
226  * \param[in] x An Eigen matrix type
227  * \param[in] d value to find
228  * \returns index of d
229  * \throws std::runtime_error if the value can't be found
230  */
231 template <typename T>
232 int findidx(T const &x, typename T::Scalar const &d,
233  typename T::Scalar const eps = std::numeric_limits<typename T::Scalar>::epsilon()) {
234  int L = 0;
235  int R = x.rows() - 1;
236  for (;;) {
237  if (L > R) {
238  // If we were already restarted crash. It doesn't make sense to
239  // increase epsilon further.
240  if (eps > std::numeric_limits<typename T::Scalar>::epsilon()) {
241  throw std::runtime_error("Search failed");
242  }
243  // Restart search with larger epsilon
244  fprintf(stderr, "WARNING: Restarting search with %.16f\n%s:%d in %s\n", 2 * eps,
245  __FILE__, __LINE__, __func__);
246  return findidx(x, d, 2 * eps);
247  }
248  int m = (L + R) / 2;
249  if (x(m) < d) {
250  L = m + 1;
251  } else if (x(m) > d) {
252  R = m - 1;
253  } else {
254  return m;
255  }
256  // x(m) and d might not actually be equal in the floating point sense.
257  // We do fuzzy compare to account for that.
258  if (std::abs(x(m) - d) <= std::max(std::abs(x(m)), std::abs(d)) * eps) {
259  return m;
260  }
261  }
262 }

264 /** \brief Compute radial matrix elements
265  *
266  * The radial matrix element can be calculated from the integral
267  * \f[
268  * \langle nlj| \hat{p}^{\mathrm{rad}}_\kappa |n'l'j' \rangle =
269  * \int \Psi^{\text{rad}}_{nlj}(r) \Psi^{\text{rad}}_{n'l'j'}(r)
270  * r^{2+\kappa} \; dr \f] The part \f$ r^{2+\kappa} \f$ varies with rescaling
271  * the domain and is this given by the power_kernel function.
272  *
273  * \tparam T Method for calculating the wavefunction
274  * \param[in] qd1 Quantum defect data for first atom
275  * \param[in] power Exponent kappa
276  * \param[in] qd2 Quantum defect data for second atom
277  * \returns Radial matrix element
278  */
279 template <typename T>
280 double IntegrateRadialElement(QuantumDefect const &qd1, int power, QuantumDefect const &qd2) {
281  T N1(qd1);
282  T N2(qd2);
283 
284  auto const &xy1 = N1.integrate();
285  auto const &xy2 = N2.integrate();
286  auto const dx = T::dx;
287 
288  auto const xmin = xy1(0, 0) >= xy2(0, 0) ? xy1(0, 0) : xy2(0, 0);
289  auto const xmax = xy1(xy1.rows() - 1, 0) <= xy2(xy2.rows() - 1, 0) ? xy1(xy1.rows() - 1, 0)
290  : xy2(xy2.rows() - 1, 0);
291 
292  double mu = 0;
293  // If there is an overlap, calculate the matrix element
294  if (xmin <= xmax) {
295  int start1 = findidx(xy1.col(0), xmin);
296  int end1 = findidx(xy1.col(0), xmax);
297  int start2 = findidx(xy2.col(0), xmin);
298  int end2 = findidx(xy2.col(0), xmax);
299 
300  int i1, i2;
301  for (i1 = start1, i2 = start2; i1 < end1 && i2 < end2; ++i1, ++i2) {
302  mu += xy1(i1, 1) * xy2(i2, 1) * std::pow(xy1(i1, 0), T::power_kernel(power)) * dx;
303  }
304  mu = 2 * mu;
305  }
306 
307  // The radial matrix element is returned in atomic units
308  return mu;
309 }

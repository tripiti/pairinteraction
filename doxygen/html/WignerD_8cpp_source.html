39 double WignerD::evalWignerdPiHalf(float j, float m, float mp) {
40  double r = 0;
41  for (unsigned int k = std::max(0, static_cast<int>(mp - m)); k <= j + std::min(mp, -m); ++k) {
42  r += std::pow(-1, k) * boost::math::binomial_coefficient<double>(j + mp, k) *
43  boost::math::binomial_coefficient<double>(j - mp, k + m - mp);
44  }
45  r *= std::pow(-1., m - mp) / std::pow(2., j) *
46  std::sqrt(
47  boost::math::factorial<double>(j + m) * boost::math::factorial<double>(j - m) /
48  (boost::math::factorial<double>(j + mp) * boost::math::factorial<double>(j - mp)));
49  return r;
50 }

52 double WignerD::evalWignerd(float j, float m, float mp, double beta) {
53  std::complex<double> r = 0;
54  for (float mpp = j; mpp >= -j; --mpp) { // TODO parallelize
55  r += this->evalWignerdPiHalf(j, m, mpp) *
56  std::complex<double>(std::cos(-mpp * beta), std::sin(-mpp * beta)) *
57  this->evalWignerdPiHalf(j, mpp, -mp);
58  }
59  r *= std::pow(std::complex<double>(0, 1), 2. * j - m - mp) * std::pow(-1., 2. * m);
60  return r.real();
61 }

39 void HamiltonianOne::changeToSpherical(double val_x, double val_y, double val_z, double &val_p,
40  double &val_m, double &val_0) {
41  if (val_y != 0) {
42  std::string msg("For fields with non-zero y-coordinates, a complex data type is needed.");
43  std::cout << boost::format(">>ERR%s") % msg << std::endl;
44  throw std::runtime_error(msg);
45  }
46  val_p = -val_x / std::sqrt(2);
47  val_m = val_x / std::sqrt(2);
48  val_0 = val_z;
49 }

33 /** \brief Quantum defect storage
34  *
35  * The quantum defect queries a database for Rydberg-Ritz coefficients and
36  * model potential parameters. The Rydberg-Ritz coefficients are not exposed to
37  * the outside but included in the energy data member.
38  *
39  * The slight deviation from the hydrogen atom for Rydberg atoms are captured
40  * in the so-called quantum defect. The hydrogen energy is given by \f$ E =
41  * \mathrm{Ry}/n^2 \f$. For Rydberg atoms this is slightly modified to \f$ E^*
42  * = \mathrm{Ry}/n^{*2} \f$ where \f$ n^* = n - \delta_{nlj} \f$. This
43  * correction can be expanded in a series
44  * \f[
45  * \delta_{nlj} = \delta_0
46  * + \frac{\delta_2}{(n-\delta_0)^2}
47  * + \frac{\delta_4}{(n-\delta_0)^4}
48  * + \frac{\delta_6}{(n-\delta_0)^6}
49  * + \cdots \;
50  * \f]
51  * The coefficients of the series \f$ \delta_i \f$ (Rydberg-Ritz coefficients)
52  * are loaded from the database.
53  *
54  * The object also loads the parameters for the model potentials which are used
55  * in the Numerov object.
56  */

59  /** \brief Key type for quantum defect cache */

66  /** \brief Comparison for keys */
67  bool operator==(Key const &o) const {
68  return (species == o.species && n == o.n && l == o.l && j == o.j);
69  }
70  };

72  //** \brief Hash for Key */
73  struct Hash {
74  /** \brief Hash the key using boost::hash_combine */
75  std::size_t operator()(Key const &key) const {
76  std::size_t seed = 0;
77  boost::hash_combine(seed, key.species);
78  boost::hash_combine(seed, key.n);
79  boost::hash_combine(seed, key.l);
80  boost::hash_combine(seed, key.j);
81  return seed;
82  }
83  };

73 /** \brief SQLite Statement handle
74  *
75  * This object wraps an SQLite statement. It is iterable in a while
76  * loop with the step() function or in a range-based for loop.

107  */

125  /** \brief Prepare the statement
126  *
127  * Prepare the statement for stepping.
128  *
129  * \throws sqlite::error
130  */
131  void prepare() {
132  sqlite3_stmt *pStmt; // is managed below
133  auto err = sqlite3_prepare_v2(m_db, m_sql.c_str(), -1, &pStmt, nullptr);
134  m_stmt.reset(pStmt);
135 
136  handle_error(err);
137 
138  m_prepared = true;
139  }
140 
141  /** \brief Step the statement
142  *
143  * This steps the statement.
144  *
145  * \returns true if there is a row, false otherwise
146  *
147  * \throws sqlite::error
148  */
149  bool step() {
150  if (!m_prepared) {
151  handle_error(SQLITE_MISUSE);
152  }
153  if (!m_valid) {
154  handle_error(SQLITE_DONE);
155  }
156 
157  auto err = sqlite3_step(m_stmt.get());
158 
159  switch (err) {
160  case SQLITE_DONE:
161  m_valid = false;
162  return false;
163  case SQLITE_ROW:
164  m_valid = true;
165  return true;
166  }
167 
168  m_valid = false;
169  handle_error(err);
170  return false;
171  }
172 
173  /** \brief Reset the statement
174  *
175  * Reset the statement so you can rebind values.
176  *
177  * \throws sqlite::error
178  */
179  void reset() {
180  handle_error(sqlite3_reset(m_stmt.get()));
181  m_valid = true;
182  }

184  /** \brief Execute SQLite statements
185  *
186  * The SQL statements are passed to this function as a string and
187  * are executed in-place.
188  *
189  * \param[in] sql SQL statements
190  * \throws sqlite::error
191  */
192  void exec(std::string const &sql) {
193  set(sql);
194  auto err = sqlite3_exec(m_db, m_sql.c_str(), nullptr, nullptr, nullptr);

196  }

253  /** \brief Statement iterator
254  *
255  * The iterator builds upon Boost Iterator Facade to implement an
256  * iterator over the statement with minimal boilerplate. We refer
257  * to the official documentation for the internal functions:
258  * http://www.boost.org/libs/iterator/
259  */
260  class iterator
261  : public boost::iterator_facade<iterator, statement, boost::forward_traversal_tag> {
262  private:
263  friend class boost::iterator_core_access;
264  statement *m_stmt;
265  bool m_done;
266  bool m_end;
267 
268  void increment() {
269  if (!m_done) {
270  m_done = m_stmt->step();
271  } else {
272  m_end = true;
273  }
274  }
275 
276  bool equal(iterator const & /*unused*/) const { return m_end; }
277 
278  statement &dereference() const {
279 #ifndef NDEBUG
280  if (m_end)
281  throw std::out_of_range("iterator out of range");
282 #endif
283  return *m_stmt;
284  }
285 
286  public:
287  iterator() : m_stmt{nullptr}, m_done{true}, m_end{true} {}
288 
289  explicit iterator(statement *p) : m_stmt{p}, m_done{false}, m_end{false} { m_stmt->step(); }
290  };

319 /** \brief SQLite Database handle
320  *
321  * This object handles the connection to the underlying SQLite databse. It
322  * provides a high-level object oriented RAII interface to the C-bindings of
323  * SQLite3.
324  */
325 class handle final {
326  std::unique_ptr<sqlite3, decltype(&sqlite3_close)> m_db;
327  int m_threshold;
328 
329  static int busy_handler(void *self, int num_prior_calls) {
330  int thresh = static_cast<handle *>(self)->m_threshold;
331  // Sleep if handler has been called less than num_prior_calls
332  if (num_prior_calls < thresh) {
333  std::this_thread::sleep_for(std::chrono::microseconds(utils::randint(2000, 20000)));
334  return 1;
335  }
336 
337  return 0; // Make sqlite3 return SQLITE_BUSY
338  }
339 
340  void install_busy_handler() {
341  auto err = sqlite3_busy_handler(m_db.get(), busy_handler, this);
342 
343  if (err != 0) {
344  throw error(err, sqlite3_errmsg(m_db.get()));
345  }
346  }
347 
348 public:
349  /** \brief Conversion operator
350  *
351  * The handle object implicitly behaves like the underlying
352  * pointer.
353  *
354  * \returns the raw database pointer
355  */
356  operator sqlite3 *() { return m_db.get(); }
357 
358  /** \brief Constructor
359  *
360  * It takes a filename and a bitmask of parameters to open the
361  * database connection. This is useful if you have to open the
362  * connection, e.g., read-only. By default it uses the standard
363  * flags for opening a database connection. See
364  * https://www.sqlite.org/c3ref/open.html for details.
365  *
366  * \param[in] filename fully-qualified filename of the database
367  * \param[in] flags options to open the database (optional)
368  * \throws sqlite::error
369  */
370  explicit handle(std::string const &filename,
371  int flags = SQLITE_OPEN_READWRITE | SQLITE_OPEN_CREATE)
372  : m_db{nullptr, sqlite3_close}, m_threshold{100000} {
373  sqlite3 *tmp_db;
374  auto err = sqlite3_open_v2(filename.c_str(), &tmp_db, flags, nullptr);
375  m_db.reset(tmp_db);
376 
377  if (err != 0) {
378  throw error(err, sqlite3_errmsg(m_db.get()));
379  }
380 
381  install_busy_handler();
382  }
383 
384  /** \brief Move constructor
385  *
386  * Reinstalls the busy handler.
387  */
388  handle(handle &&other) noexcept : m_db{std::move(other.m_db)}, m_threshold{other.m_threshold} {
389  install_busy_handler();
390  }
391 
392  /** \brief Move assignment operator
393  *
394  * Reinstalls the busy handler.
395  */
396  handle &operator=(handle &&other) noexcept {
397  if (this != std::addressof(other)) // skip self-assignment
398  {
399  m_db = std::move(other.m_db);
400  m_threshold = other.m_threshold;
401  install_busy_handler();
402  }
403  return *this;
404  }
405 };

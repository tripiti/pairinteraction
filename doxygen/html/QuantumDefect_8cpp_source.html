78 void QuantumDefect::setup(sqlite3 *db) {
79  static Cache<Key, Element, Hash> cache;
80 
81  Key const key{species, n, l, j};
82  if (auto oe = cache.restore(key)) {
83  e = oe.get(); // Restore cache
84  return; // Return early
85  }
86 
87  std::stringstream ss;
88  sqlite::statement stmt(db);
89  int pot_max_l, ryd_max_l;
90  int pot_l, ryd_l;
91  double ryd_max_j;
92  double ryd_j;
93 
94  // Determine maximal L for model potentials
95  stmt.set("select MAX(L) from model_potential where (element = ?);");
96  stmt.prepare();
97  stmt.bind(1, species);
98  if (stmt.step()) {
99  pot_max_l = stmt.get<int>(0);
100  } else {
101  throw no_potential(*this);
102  }
103 
104  // The l to be used is the minimum of the two below
105  pot_l = std::min(l, pot_max_l);
106 
107  // Determine maximal L for Rydberg-Ritz coefficients
108  stmt.reset();
109  stmt.set("select MAX(L) from rydberg_ritz where (element = ?);");
110  stmt.prepare();
111  stmt.bind(1, species);
112  if (stmt.step()) {
113  ryd_max_l = stmt.get<int>(0);
114  } else {
115  throw no_defect(*this);
116  }
117 
118  // The l to be used is the minimum of the two below
119  ryd_l = std::min(l, ryd_max_l);
120 
121  // Determine maximal J for Rydberg-Ritz coefficients
122  stmt.reset();
123  stmt.set("select MAX(J) from rydberg_ritz where ((element = ?1) and (L = "
124  "?2));");
125  stmt.prepare();
126  stmt.bind(1, species);
127  stmt.bind(2, ryd_l);
128  if (stmt.step()) {
129  ryd_max_j = stmt.get<double>(0);
130  } else {
131  throw no_defect(*this);
132  }
133 
134  // The j to be used is the minimum of the two below
135  ryd_j = std::min(j, ryd_max_j);
136 
137  // Load model potentials from database
138  stmt.reset();
139  stmt.set("select ac,Z,a1,a2,a3,a4,rc from model_potential where ((element "
140  "= ?1) and (L = ?2));");
141  stmt.prepare();
142  stmt.bind(1, species);
143  stmt.bind(2, pot_l);
144  if (stmt.step()) {
145  e.ac = stmt.get<double>(0);
146  e.Z = stmt.get<int>(1);
147  e.a1 = stmt.get<double>(2);
148  e.a2 = stmt.get<double>(3);
149  e.a3 = stmt.get<double>(4);
150  e.a4 = stmt.get<double>(5);
151  e.rc = stmt.get<double>(6);
152  } else {
153  throw no_potential(*this);
154  }
155 
156  // Load Rydberg-Ritz coefficients from database
157  stmt.reset();
158  stmt.set("select d0,d2,d4,d6,d8,Ry from rydberg_ritz where ((element = ?1) "
159  "and (L = ?2) and (J = ?3));");
160  stmt.prepare();
161  stmt.bind(1, species);
162  stmt.bind(2, ryd_l);
163  stmt.bind(3, ryd_j);
164  e.nstar = n;
165  double Ry_inf = 109737.31568525;
166  double Ry;
167  if (stmt.step()) {
168  double d0 = stmt.get<double>(0);
169  double d2 = stmt.get<double>(1);
170  double d4 = stmt.get<double>(2);
171  double d6 = stmt.get<double>(3);
172  double d8 = stmt.get<double>(4);
173  Ry = stmt.get<double>(5);
174  e.nstar -= d0 + d2 / pow(n - d0, 2) + d4 / pow(n - d0, 4) + d6 / pow(n - d0, 6) +
175  d8 / pow(n - d0, 8);
176  } else {
177  throw no_defect(*this);
178  }
179 
180  e.energy = -.5 * (Ry / Ry_inf) / (e.nstar * e.nstar) * au2GHz;
181 
182  cache.save(key, e);
183 }

64  void save(bytes_t &bytes) {
65  size_t nItems = buffer_types.size();
66 
67  // clear and resize bytes
68  bytes.clear();
69  bytes.resize(size());
70 
71  // save items to bytes
72  pbytes = bytes.begin();
73 
74  for (size_t i = 0; i < nItems; ++i) {
75  if (buffer_isVector[i]) {
76  serialize(&type_ids[std::type_index(typeid(storage_idx_t))], sizeof(type_t));
77  serialize(&buffer_nums[i], sizeof(storage_idx_t));
78  }
79 
80  serialize(&buffer_types[i], sizeof(type_t));
81  serialize(buffer_pitems[i], buffer_nums[i] * buffer_sizes[i]);
82  }
83 
84  // the buffers are not needed anymore
85  buffer_isVector.clear();
86  buffer_types.clear();
87  buffer_pitems.clear();
88  buffer_nums.clear();
89  buffer_sizes.clear();
90  }

92  size_t position() {
93  size_t sz = size();
94  if (sz > 0) {
95  return sz;
96  }
97  return std::distance(cpbytes_start, cpbytes);
98  }

131  size_t size() {
132  size_t nItems = buffer_types.size();
133  size_t totalsize = 0;
134 
135  for (size_t i = 0; i < nItems; ++i) {
136  if (buffer_isVector[i]) {
137  totalsize += sizeof(type_t);
138  totalsize += sizeof(storage_idx_t);
139  }
140  totalsize += sizeof(type_t);
141  totalsize += buffer_nums[i] * buffer_sizes[i];
142  }
143 
144  return totalsize;

36 double V(QuantumDefect const &qd, double x) {
37  double Z_l =
38  1 + (qd.Z - 1) * std::exp(-qd.a1 * x) - x * (qd.a3 + qd.a4 * x) * std::exp(-qd.a2 * x);
39  double V_c = -Z_l / x;
40  double V_p = -qd.ac / (2. * x * x * x * x) * (1 - std::exp(-std::pow(x / qd.rc, 6)));
41  double V_so = 0.0;
42  if (qd.l < 4) {
43  double s = 0.5;
44  if (std::isdigit(qd.species.back()) != 0) {
45  s = ((qd.species.back() - '0') - 1) / 2.; // TODO think of a better solution
46  }
47  double alpha = 7.2973525664e-3; // ~1/137 fine-structure constant CODATA 2014
48  V_so =
49  alpha * alpha / (4 * x * x * x) * (qd.j * (qd.j + 1) - qd.l * (qd.l + 1) - s * (s + 1));
50  }
51  return V_c + V_p + V_so;
52 }

54 double g(QuantumDefect const &qd, double x) {
55  return (2. * qd.l + .5) * (2. * qd.l + 1.5) / x + 8 * x * (V(qd, x) - qd.energy / au2GHz);

60 Numerov::Numerov(QuantumDefect const &qd) : qd(qd) {
61  // augmented classical turning point
62  double xmin = qd.n * qd.n - qd.n * std::sqrt(qd.n * qd.n - (qd.l - 1) * (qd.l - 1));
63  if (xmin < 2.08) {
64  xmin = std::floor(std::sqrt(2.08));
65  } else {
66  xmin = std::floor(std::sqrt(xmin));
67  }
68 
69  double const xmax = std::sqrt(2 * qd.n * (qd.n + 15));
70  double const nsteps = std::ceil((xmax - xmin) / dx);
71 
72  xy = eigen_dense_double_t::Zero(nsteps, 2);
73 
74  for (int i = 0; i < nsteps; ++i) {
75  xy(i, 0) = (xmin + i * dx);

79 eigen_dense_double_t Numerov::integrate() {
80  using model_potential::g;
81 
82  int const nsteps = xy.rows();
83 
84  // Set the initial condition
85  if ((qd.n - qd.l) % 2 == 0) {
86  xy(nsteps - 2, 1) = -1e-10;
87  } else {
88  xy(nsteps - 2, 1) = 1e-10;
89  }
90 
91  // Perform the integration using Numerov's scheme
92  for (int i = nsteps - 3; i >= 0; --i) {
93  double A = (2. + 5. / 6. * dx * dx * g(qd, xy(i + 1, 0) * xy(i + 1, 0))) * xy(i + 1, 1);
94  double B = (1. - 1. / 12. * dx * dx * g(qd, xy(i + 2, 0) * xy(i + 2, 0))) * xy(i + 2, 1);
95  double C = 1. - 1. / 12. * dx * dx * g(qd, xy(i, 0) * xy(i, 0));
96  xy(i, 1) = (A - B) / C;
97  }
98 
99  // Normalization
100  double norm = 0;
101  for (int i = 0; i < nsteps; ++i) {
102  norm += xy(i, 1) * xy(i, 1) * xy(i, 0) * xy(i, 0) * dx;
103  }
104  norm = std::sqrt(2 * norm);
105 
106  if (norm > 0.0) {
107  for (int i = 0; i < nsteps; ++i) {
108  xy(i, 1) /= norm;
109  }
110  }
111 
112  return xy;
113 }

131 double WhittakerW(double k, double m, double z) {
132  return std::exp(-.5 * z) * std::pow(z, m + .5) * HypergeometricU(m - k + .5, 1 + 2 * m, z);
133 }

135 double RadialWFWhittaker(double r, double nu, int l) {
136  return 1 / std::sqrt(nu * nu * std::tgamma(nu + l + 1) * std::tgamma(nu - l)) *
137  WhittakerW(nu, l + .5, 2 * r / nu);

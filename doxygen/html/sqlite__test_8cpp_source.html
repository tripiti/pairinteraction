35 BOOST_AUTO_TEST_CASE(sqlite_query_test) // NOLINT
36 {
37  BOOST_CHECK_THROW(sqlite::handle db("no such database", SQLITE_OPEN_READWRITE), sqlite::error);
38  // Open an in-memory database for tests
39  sqlite::handle db(":memory:");
40 
41  sqlite::statement stmt(db);
42  BOOST_CHECK_NO_THROW(stmt.set("This is not valid SQL"));
43  BOOST_CHECK_THROW(stmt.step(), sqlite::error);
44  BOOST_CHECK_THROW(stmt.prepare(), sqlite::error);
45  BOOST_CHECK_THROW(stmt.exec("Neither is this"), sqlite::error);
46 
47  // Check string calling
48  std::string string_query("create table test(text,integer,real);");
49  BOOST_CHECK_NO_THROW(stmt.reset());
50  BOOST_CHECK_NO_THROW(stmt.set(string_query));
51  BOOST_CHECK_NO_THROW(stmt.prepare());
52  BOOST_CHECK_EQUAL(stmt.step(), false);
53 
54  // Check stringstream calling
55  std::stringstream ss_query;
56  ss_query << "insert into test values(?1,?2,?3);";
57  BOOST_CHECK_NO_THROW(stmt.reset());
58  BOOST_CHECK_NO_THROW(stmt.set(ss_query));
59  // Insert some stuff
60  BOOST_CHECK_NO_THROW(stmt.prepare());
61  BOOST_CHECK_NO_THROW(stmt.bind(1, "Hello World!"));
62  BOOST_CHECK_NO_THROW(stmt.bind(2, 1729));
63  BOOST_CHECK_NO_THROW(stmt.bind(3, 0.5));
64  BOOST_CHECK_EQUAL(stmt.step(), false);
65  // Reuse the query set above
66  BOOST_CHECK_NO_THROW(stmt.reset());
67  BOOST_CHECK_NO_THROW(stmt.prepare());
68  BOOST_CHECK_NO_THROW(stmt.bind(1, "Goodbye Earth!"));
69  BOOST_CHECK_NO_THROW(stmt.bind(2, 42));
70  BOOST_CHECK_NO_THROW(stmt.bind(3, 1.125));
71  BOOST_CHECK_EQUAL(stmt.step(), false);
72  BOOST_CHECK_THROW(stmt.step(), sqlite::error);
73 
74  // Check result
75  BOOST_CHECK_NO_THROW(stmt.reset());
76  BOOST_CHECK_NO_THROW(stmt.set("select * from test;"));
77  BOOST_CHECK_NO_THROW(stmt.prepare());
78  BOOST_CHECK_EQUAL(stmt.step(), true);
79  BOOST_CHECK_EQUAL(stmt.get<std::string>(0), "Hello World!");
80  BOOST_CHECK_EQUAL(stmt.get<int>(1), 1729);
81  BOOST_CHECK_EQUAL(stmt.get<double>(2), 0.5);
82  BOOST_CHECK_EQUAL(stmt.step(), true);
83  BOOST_CHECK_EQUAL(stmt.get<std::string>(0), "Goodbye Earth!");
84  BOOST_CHECK_EQUAL(stmt.get<int>(1), 42);

86  BOOST_CHECK_EQUAL(stmt.step(), false);
87
